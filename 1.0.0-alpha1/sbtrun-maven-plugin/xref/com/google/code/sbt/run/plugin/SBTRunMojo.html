206     protected Java prepareAntJavaTask( boolean fork )
207         throws ArtifactResolutionException, ArtifactNotFoundException, InvalidDependencyVersionException, ProjectBuildingException//, IOException
208     {
209         File baseDir = project.getBasedir();
210 
211         Project antProject = createProject();
212         Path classPath = getProjectClassPath( antProject );
213 
214         Java javaTask = new Java();
215         javaTask.setTaskName( "sbt" );
216         javaTask.setProject( antProject );
217         javaTask.setClassname( "xsbt.boot.Boot" );
218         javaTask.setClasspath( classPath );
219         javaTask.setFork( fork );
220         javaTask.createArg().setLine( args );
221         if ( fork )
222         {
223             javaTask.setDir( baseDir );
224 
225             if ( jvmArgs != null )
226             {
227                 jvmArgs = jvmArgs.trim();
228                 if ( jvmArgs.length() > 0 )
229                 {
230                     String[] jvmArgsArray = jvmArgs.split( " " );
231                     for ( String arg : jvmArgsArray )
232                     {
233                         javaTask.createJvmarg().setValue( arg );
234                         getLog().debug( "  Adding jvmarg '" + arg + "'" );
235                     }
236                 }
237             }
238         }
239         else
240         {
241             // find and add all system properties in "jvmArgs"
242             if ( jvmArgs != null )
243             {
244                 jvmArgs = jvmArgs.trim();
245                 if ( jvmArgs.length() > 0 )
246                 {
247                     String[] jvmArgsArray = jvmArgs.split( " " );
248                     for ( String arg : jvmArgsArray )
249                     {
250                         if ( arg.startsWith( "-D" ) )
251                         {
252                             arg = arg.substring( 2 );
253                             int p = arg.indexOf( '=' );
254                             if ( p >= 0 )
255                             {
256                                 String key = arg.substring( 0, p );
257                                 String value = arg.substring( p + 1 );
258                                 getLog().debug( "  Adding system property '" + arg + "'" );
259                                 addSystemProperty( javaTask, key, value );
260                             }
261                             /*else
262                             {
263                                 // TODO - throw an exception
264                             }*/
265                         }
266                     }
267                 }
268             }
269         }
270         return javaTask;
271     }

324     private Artifact getResolvedArtifact( String groupId, String artifactId, String version )
325         throws ArtifactNotFoundException, ArtifactResolutionException
326     {
327         Artifact artifact = factory.createArtifact( groupId, artifactId, version, Artifact.SCOPE_RUNTIME, "jar" );
328         resolver.resolve( artifact, remoteRepos, localRepo );
329         return artifact;
330     }

332     private Set<Artifact> getAllDependencies( Artifact artifact )
333         throws ArtifactNotFoundException, ArtifactResolutionException, InvalidDependencyVersionException,
334         ProjectBuildingException
335     {
336         Set<Artifact> result = new HashSet<Artifact>();
337         MavenProject p = mavenProjectBuilder.buildFromRepository( artifact, remoteRepos, localRepo );
338         Set<Artifact> d = resolveDependencyArtifacts( p );
339         result.addAll( d );
340         for ( Artifact dependency : d )
341         {
342             Set<Artifact> transitive = getAllDependencies( dependency );
343             result.addAll( transitive );
344         }
345         return result;
346     }

348     /**
349      * This method resolves the dependency artifacts from the project.
350      * 
351      * @param theProject The POM.
352      * @return resolved set of dependency artifacts.
353      * @throws ArtifactResolutionException
354      * @throws ArtifactNotFoundException
355      * @throws InvalidDependencyVersionException
356      */
357     private Set<Artifact> resolveDependencyArtifacts( MavenProject theProject )
358         throws ArtifactNotFoundException, ArtifactResolutionException, InvalidDependencyVersionException
359     {
360         AndArtifactFilter filter = new AndArtifactFilter();
361         filter.add( new ScopeArtifactFilter( Artifact.SCOPE_TEST ) );
362         filter.add( new NonOptionalArtifactFilter() );
363         // TODO follow the dependenciesManagement and override rules
364         Set<Artifact> artifacts = theProject.createArtifacts( factory, Artifact.SCOPE_RUNTIME, filter );
365         for ( Artifact artifact : artifacts )
366         {
367             resolver.resolve( artifact, remoteRepos, localRepo );
368         }
369         return artifacts;
370     }

372     private static class NonOptionalArtifactFilter
